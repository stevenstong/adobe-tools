377 template<class T, class S> ErrorCode Accessor<T,S>::Access( IScriptRequestData* data, IScript* script, const ScriptID& propID )
378 {
379  ErrorCode err;
380  ScriptData scriptData;
381  if (data->IsPropertyGet())
382  {
383  InterfacePtr<const IAttrReport> attrReport;
384  err = fScriptUtils->GetAttributeReport(fAttrClass, attrReport);
385  if (err == kUnSetError)
386  {
387  scriptData.SetEnumeration(en_Nothing);
388  data->AppendReturnData( script, propID, scriptData ) ;
389  return kSuccess;
390  }
391 
392  if (err != kSuccess) return err;
393 
394  InterfacePtr<const T> attr(attrReport, UseDefaultIID());
395  if (!attr) return kFailure;
396 
397  return fMap.GetForGetProperty( script, data, propID, attr->Get() ) ;
398  }
399  else if (data->IsPropertyPut())
400  {
401  err = data->ExtractRequestData(propID, scriptData);
402  if (err != kSuccess) return err;
403 
404  if (Utils<IScriptUtils>()->IsNothing( scriptData )) // removing an attribute
405  {
406  err = fScriptUtils->RemoveAttribute( fAttrClass ) ;
407  }
408  else // adding/modifying an attribute
409  {
410  int32 val(0);
411  err = fMap.GetForSetProperty( script, data, propID, val ) ;
412  if (err != kSuccess) return err;
413 
414  InterfacePtr<T> attrR(::CreateObject2<T>(fAttrClass));
415  if (!attrR) return kFailure;
416 
417  attrR->Set(static_cast<S>(val));
418  err = fScriptUtils->AddAttribute( attrR ) ;
419  }
420 
421  return err ;
422  }
423 
424  return kFailure ;
425 }
