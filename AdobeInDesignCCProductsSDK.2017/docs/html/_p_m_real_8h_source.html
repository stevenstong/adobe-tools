48 // MSFT VC is quite good at inlining things and produces code nearly
49 // identical to using doubles.

53 // expanded. There are other compiler back ends for MW that we could

58 // These NaN macros should be okay on any IEEE 754 compatible machine and the hexidecimal constant
59 // is a valid signalling NaN. Note that __isSNaN returns true only for THIS particular signalling
60 // NaN and doesn't detect all possible. You can't use std::numeric_limits<double>::signaling_NaN()
61 // here because when the value is loaded and then stored the FPU turns it into a quiet NaN (because
62 // of our FPU flags). This solution now means that in the debug build, uninitialized PMReals contain
63 // a signalling NaN, that usage of such a PMReal causes an assert, but generating a quiet NaN as part
64 // of a computation and subsequently using it no longer causes an erroneous asserts.
65 
66 #define __isSNaN(d) (*((int64 *)(&d)) == 0x7ff0000000000001LL)
67 #define __setSNaN(d) do {*((int64 *)(&d)) = 0x7ff0000000000001LL; } while (0)

698  // allow epsilon compare with angles in degress (360 == 0)
699 inline bool16 AnglesEqual(const PMReal& a, const PMReal& b)
700 {
701  ASSERT_MSG(!__isSNaN(a.fRealValue),"Trying to use uninitialized PMReal");
702  ASSERT_MSG(!__isSNaN(b.fRealValue),"Trying to use uninitialized PMReal");
703  if (a == b)
704  return kTrue;
705  else if (a > b)
706  return ::fabs((a.fRealValue - 360.0) - b.fRealValue) < kPMRealEpsilon;
707  else
708  return ::fabs(a.fRealValue - (b.fRealValue - 360.0)) < kPMRealEpsilon;
709 }
