390 template <class IteratorType>
391 inline UnicodeSavvyString& UnicodeSavvyString::assign(IteratorType b, IteratorType e, size_type nCodePoints)
392 {
393  // The value from the iterator should be 16 bit
394  typedef typename std::iterator_traits<IteratorType>::value_type iterator_value_type;
395  BOOST_STATIC_ASSERT(sizeof(iterator_value_type)*CHAR_BIT == 16);
396  typedef typename std::iterator_traits<IteratorType>::iterator_category iterator_category;
397 
398 #ifdef DEBUG
399  InvariantChecker checkThis(*this);
400 #endif
401 
402  // Dispatch to the optimized version
403  assign_impl(b, e, nCodePoints, iterator_category());
404  return *this;
405 }

407 // Input iterators assign implementation
408 template <class InputIterator>
409 inline void UnicodeSavvyString::assign_impl(InputIterator b, InputIterator e, size_type nCodePoints, std::input_iterator_tag)
410 {
411  UnicodeSavvyString temp;
412 
413  // Try to optimize allocations in case the client told us the number of code points.
414  if (nCodePoints)
415  {
416  temp.reserve(nCodePoints);
417  }
418 
419  std::copy(b, e, std::back_inserter(temp));
420  swap(*this, temp);
421 }
422 
423 template <class FwdIterator>
424 inline void UnicodeSavvyString::assign_impl(FwdIterator b, FwdIterator e, size_type nCodePoints, std::forward_iterator_tag)
425 {
426  const difference_type nCodeValues = std::distance(b, e);
427  ASSERT(nCodeValues >= 0);
428 
429  // Make room for the chars
430  code_value* buffer = GetBufferForWriting(nCodeValues);
431 
432  // Copy them into the buffer
433  std::copy(b, e, buffer);
434 
435  // Ensure terminating null
436  buffer[nCodeValues] = code_value();
437 
438  fUTF16BufferLength = nCodeValues;
439 
440  // Calculate how many code points we will have at the end
441  if (nCodePoints == 0 && nCodeValues)
442  {
443  fNumChars = CountCharsUtil(buffer, nCodeValues);
444  }
445  else
446  {
447  ASSERT((size_type)CountCharsUtil(buffer, nCodeValues) == nCodePoints);
448  fNumChars = nCodePoints;
449  }
450 }
