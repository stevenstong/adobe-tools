71 template<class T>
72 class IDType
73 {
74  public:
75  typedef base_type data_type;
76  typedef int32 difference_type;
77  typedef uint32 value_type;
78 
79  constexpr IDType() : fIDInt(0) {}
80  //~IDType() {}
81  constexpr IDType(value_type sourceInt) : fIDInt(sourceInt) {}
82  IDType& operator=(const value_type& sourceInt) {fIDInt = sourceInt; return *this;}
83 
84  bool operator>(const IDType& rhs) const { return fIDInt > rhs.fIDInt; }
85  bool operator>(const value_type& rhs) const { return fIDInt > rhs; }
86  friend bool operator>(const value_type& lhs, const IDType<T>& rhs) { return lhs > rhs.fIDInt; }
87 
88  bool operator<(const IDType& rhs) const { return fIDInt < rhs.fIDInt; }
89  bool operator<(const value_type& rhs) const { return fIDInt < rhs; }
90  friend bool operator<(const value_type& lhs, const IDType<T>& rhs) { return lhs < rhs.fIDInt; }
91 
92  bool operator<=(const IDType& rhs) const { return !(fIDInt > rhs.fIDInt); }
93  bool operator<=(const value_type& rhs) const { return !(fIDInt > rhs); }
94  friend bool operator<=(const value_type& lhs, const IDType<T>& rhs) { return !(lhs > rhs.fIDInt); }
95 
96  bool operator>=(const IDType& rhs) const { return !(fIDInt < rhs.fIDInt); }
97  bool operator>=(const value_type& rhs) const { return !(fIDInt < rhs); }
98  friend bool operator>=(const value_type& lhs, const IDType<T>& rhs) { return !(lhs < rhs.fIDInt); }
99 
100  bool operator==(const IDType& rhs) const { return fIDInt == rhs.fIDInt; }
101  friend bool operator==(const value_type& lhs, const IDType<T>& rhs) { return lhs == rhs.fIDInt; }
102 
103  bool operator!=(const IDType& rhs) const { return !(fIDInt == rhs.fIDInt); }
104  bool operator!=(const value_type& rhs) const { return !(fIDInt == rhs); }
105  friend bool operator!=(const value_type& lhs, const IDType<T>& rhs) { return !(lhs == rhs.fIDInt); }
106 
107  IDType& operator+=(const difference_type& rhs) { fIDInt += rhs; return *this; }
108  friend IDType<T> operator+(const IDType<T>& lhs, const difference_type& rhs) { return IDType<T>(lhs) += rhs; }
109  friend IDType<T> operator+(const IDType<T>& lhs, const IDType<T>& rhs) { return IDType<T>(lhs) += rhs.fIDInt; }
110  friend IDType<T> operator+(const difference_type& lhs, const IDType<T> rhs) { return IDType<T>(rhs) += lhs; }
111 
112  // prefix
113  IDType& operator++() { *this += 1; return *this; }
114  // postfix
115  const IDType operator++(int) { IDType oldValue = *this; ++(*this); return oldValue; }
116 
117  IDType& operator-=(const IDType& rhs) { fIDInt -= rhs.fIDInt; return *this; }
118  IDType& operator-=(const difference_type& rhs) { fIDInt -= rhs; return *this; }
119  friend IDType<T> operator-(const difference_type& lhs, IDType<T> rhs) { return IDType<T>(rhs) -= lhs; }
120  friend difference_type operator-(const IDType<T>& lhs, const difference_type& rhs) { return lhs.fIDInt - rhs; }
121  friend difference_type operator-(const IDType<T>& lhs, const IDType<T>& rhs) { return lhs.fIDInt - rhs.fIDInt; }
122 
123  // prefix
124  IDType& operator--() { *this -= 1; return *this; }
125  // postfix
126  const IDType operator--(int) { IDType oldValue = *this; --(*this); return oldValue; }
127 
128  value_type operator>>(const int& rhs) const { return fIDInt >> rhs; }
129  value_type operator<<(const int& rhs) const { return fIDInt << rhs; }
130 
131  value_type operator&(const value_type& rhs) const { return fIDInt & rhs; }
132  friend value_type operator&(const value_type& lhs, const IDType<T>& rhs) { return IDType<T>(rhs) & lhs; }
133 
134  value_type operator|(const value_type& rhs) { return fIDInt | rhs; }
135  friend value_type operator|(const value_type& lhs, const IDType<T>& rhs) { return IDType<T>(rhs) | lhs; }
136 
137  IDType& operator|=(const value_type& rhs) { fIDInt |= rhs; return *this; }

139

141  inline const value_type& Get() const { return fIDInt; }
142
