958 template <>
959 class InterfacePtr<IPMFont>
960 {
961 public:
962  // Note: Constructing an InterfacePtr from an existing,
963  // "real" interface pointer implies the InterfacePtr
964  // now owns the reference that was held by the real pointer.
965  InterfacePtr(IPMFont * pFace)
966  : fFace(pFace) {}
967 
968  ~InterfacePtr()
969  { Release_(fFace); }
970 
971  InterfacePtr(void) : fFace(nil)
972  {}
973 
974  InterfacePtr(const InterfacePtr<IPMFont>& f) : fFace(AddRef_(f.get()))
975  {}
976 
977  // Constructor that enables InterfacePtr<const IFoo> to be constructed from an InterfacePtr<IFoo>.
978 // template <class OtherType>
979 // InterfacePtr(const InterfacePtr<OtherType>& p)
980 // :fFace(AddRef_(p.get())) { }
981 
982  InterfacePtr<IPMFont>& operator=(const InterfacePtr<IPMFont>& p)
983  { reset(AddRef_(p.get())); return *this; }
984  InterfacePtr<IPMFont>& operator=(const InterfacePtr<const IPMFont>& p)
985  { reset(AddRef_(p.get())); return *this; }
986 
987  operator IPMFont*() const
988  { return fFace; }
989  // We allow the cast operator to return a nil pointer
990  // since it is valid to test against nil and may be valid
991  // to pass a nil pointer to some methods.
992 
993 
994  IPMFont* operator ->() const
995  {
996  #ifdef DEBUG
997  ASSERT_MSG(fFace != nil,"About to use nil interface ptr!");
998  #endif
999  return fFace;
1000  }
1001 
1002  // Added forget, get and reset from InterfacePtr
1003  IPMFont* forget()
1004  {
1005  IPMFont* result = fFace;
1006  fFace = 0;
1007  return result;
1008  }
1009 
1010  IPMFont* get() const
1011  { return fFace; }
1012 
1013  void reset(IPMFont* p = 0)
1014  {
1015  Release_(fFace);
1016  fFace = p;
1017  }
1018 
1019  void reset(const IPMFont* p)
1020  {
1021  Release_(fFace);
1022  fFace = const_cast<IPMFont*>(p);
1023  }
1024 
1025 private:
1026  static IPMFont* AddRef_(const IPMFont* p)
1027  { if(p) p->AddRef(); return const_cast<IPMFont*>(p);}
1028  static void Release_(IPMFont* p)
1029  { if(p) p->Release(); }
1030 
1031  IPMFont * fFace;
1032 };
