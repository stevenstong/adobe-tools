344  Font_Info(const CTFontDict* ctFontDict, int32 numDesignVectors = 0, Fixed* designVectors = nil, uint32 flags = 0)
345  : fctFontDict(ctFontDict),
346  fnumDesignVectors(numDesignVectors),
347  fFlags(flags)
348  {
349  int32 i;
350  for (i = 0; i < 4; i++)
351  fDesignVectors[i] = 0;
352  if (fnumDesignVectors && designVectors)
353  {
354  ASSERT(numDesignVectors <= 4);
355  if (fnumDesignVectors > 4)
356  fnumDesignVectors = 4;
357  for (i = 0; i < fnumDesignVectors; i++)
358  fDesignVectors[i] = designVectors[i];
359  }
360  }
361 
365  const CTFontDict* GetCTFontDict() const { return fctFontDict; }
369  size_t GetNumDesignVectors() const { return fnumDesignVectors; }
374  void GetDesignVectors(Fixed* dV) const { dV[0] = fDesignVectors[0];
375  dV[1] = fDesignVectors[1];
376  dV[2] = fDesignVectors[2];
377  dV[3] = fDesignVectors[3];
378  }
382  uint32 GetFlags() const { return fFlags; }
383 
386  Font_Info &operator= (const Font_Info& other)
387  {
388  if (this != &other)
389  {
390  fctFontDict = other.fctFontDict;
391  fnumDesignVectors = other.fnumDesignVectors;
392  Fixed fiDV[4];
393  other.GetDesignVectors(&fiDV[0]);
394  fDesignVectors[0] = fiDV[0];
395  fDesignVectors[1] = fiDV[1];
396  fDesignVectors[2] = fiDV[2];
397  fDesignVectors[3] = fiDV[3];
398  fFlags = other.fFlags;
399  }
400  return *this;
401  }
402 
406  bool16 operator==(const Font_Info& fi) const
407  {
408  if (fctFontDict != fi.GetCTFontDict())
409  return kFalse;
410  if (fnumDesignVectors != fi.GetNumDesignVectors())
411  return kFalse;
412  Fixed fiDV[4];
413  fi.GetDesignVectors(&fiDV[0]);
414  for (int32 i = 0; i < fnumDesignVectors; i++)
415  {
416  if (fDesignVectors[i] != fiDV[i])
417  return kFalse;
418  }
419  if (fFlags != fi.GetFlags())
420  return kFalse;
421 
422  return kTrue;
423  }
